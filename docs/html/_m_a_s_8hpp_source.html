226  void ComputeGoodnessOfFit() {
227 
228  variable f;
229  atl::Variable<REAL_T>::tape.Reset();
230  atl::Variable<REAL_T>::tape.recording = true;
231  atl::Variable<REAL_T>::tape.derivative_trace_level = atl::SECOND_ORDER_REVERSE;
232  this->Run(f);
233  atl::Variable<REAL_T>::tape.AccumulateSecondOrder();
234  typename mas::Information<REAL_T>::fleet_iterator fit;
235  typename mas::Information<REAL_T>::survey_model_iterator sit;
236 
237 
238  for (sit = info.survey_models.begin(); sit != info.survey_models.end(); ++sit) {
239  (*sit).second->ComputeGoodnessOfFit();
240  this->chi_squared += (*sit).second->chi_squared;
241  this->g_test += (*sit).second->g_test;
242  this->rmse += (*sit).second->rmse;
243  this->rmsle += (*sit).second->rmsle;
244  }
245 
246  for (fit = info.fleets.begin(); fit != info.fleets.end(); ++fit) {
247  (*fit).second->ComputeGoodnessOfFit();
248  this->chi_squared += (*fit).second->chi_squared;
249  this->g_test += (*fit).second->g_test;
250  this->rmse += (*fit).second->rmse;
251  this->rmsle += (*fit).second->rmsle;
252  this->AIC += (*fit).second->AIC;
253  this->BIC += (*fit).second->BIC;
